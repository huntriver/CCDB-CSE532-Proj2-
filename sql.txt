CREATE TABLE "person"
(
	PId varchar(5) NOT NULL ,
	Name varchar(20) NOT NULL,
	Addr varchar(100),
	DOB varchar(10),
	PRIMARY KEY(PId),
	CHECK (PId LIKE 'p%'),
	CHECK (dob LIKE '[0-1][1-9][/][0-3][0-9][/][1-2][0,9][0-9][0-9]')
);

CREATE TABLE "organization"
(
  	OId varchar(5) NOT NULL ,
  	Name varchar(50) NOT NULL,
  	Addr varchar(100),
  	PRIMARY KEY(OId),
  	CHECK (OId LIKE 'o%')
);

CREATE TABLE "signer"
(
	OId varchar(5) NOT NULL,
	PId varchar(5) NOT NULL,
	PRIMARY KEY(OId,PId),
	FOREIGN KEY (PId) REFERENCES person(PId),
	FOREIGN KEY (OId) REFERENCES organization(OId),
	CHECK (OId LIKE 'o%'),
	CHECK (PId LIKE 'p%')
);

CREATE TABLE "card"
(
	CId varchar(5) NOT NULL,
	OwnerID varchar(5) NOT NULL,
	Bal INT NOT NULL,
	Lim INT NOT NULL,
	PRIMARY KEY (CId),
	FOREIGN KEY (OwnerID) REFERENCES person(PId),
	FOREIGN KEY (OwnerID) REFERENCES organization(OId),
	CHECK (CId LIKE 'c%'),
	CHECK (Bal>=0),
	CHECK (Lim>=0)
);

CREATE TABLE "authorized" 
(
	PId varchar(5) NOT NULL,
	CId varchar(5) NOT NULL,
	PRIMARY KEY(PId,CId),
	CHECK (PId LIKE 'p%'),
	CHECK (CId LIKE 'c%')
)


CREATE VIEW "authuser"  AS
	SELECT	*
	FROM	authorized
